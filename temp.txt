import javax.xml.xpath.XPath;
import javax.xml.xpath.XPathFactory;
import javax.xml.xpath.XPathExpression;
import org.w3c.dom.NodeList;
import org.w3c.dom.Node;

// ... [Other imports and XmlParser class declaration]

public List<PAResponseEntity> parseXmlToEntities(String xmlFilePath) {
    List<PAResponseEntity> entities = new ArrayList<>();

    try {
        File xmlFile = new File(xmlFilePath);
        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
        DocumentBuilder builder = factory.newDocumentBuilder();
        Document document = builder.parse(xmlFile);
        document.getDocumentElement().normalize();

        XPathFactory xpathFactory = XPathFactory.newInstance();
        XPath xpath = xpathFactory.newXPath();

        NodeList paResponseNodes = document.getElementsByTagName("PAResponse");

        for (int i = 0; i < paResponseNodes.getLength(); i++) {
            Node paResponseNode = paResponseNodes.item(i);
            PAResponseEntity entity = new PAResponseEntity();

            // Extract fields using XPath
            XPathExpression exprFirstName = xpath.compile("relative/path/to/FirstName"); // Adjust XPath as needed
            String firstName = (String) exprFirstName.evaluate(paResponseNode, javax.xml.xpath.XPathConstants.STRING);
            entity.setFirstName(firstName);

            // Repeat for other fields...

            entities.add(entity);
        }
    } catch (Exception e) {
        e.printStackTrace();
    }

    return entities;
}
